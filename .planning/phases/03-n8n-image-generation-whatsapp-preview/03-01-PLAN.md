const imageUrls = loopItems.map(item => {
  const url = item.json.data?.[0]?.url || null;
  if (url && typeof url === 'string' && url.startsWith('https://')) {
    return url.split('?')[0]; // Strip query params for clean URL
  }
  return null;
}).filter(Boolean);

if (imageUrls.length !== carousel.num_images) {
  throw new Error('Se generaron ' + imageUrls.length + ' imagenes, se esperaban ' + carousel.num_images + '. Faltan slides.');
}

return [{
  json: {
    instagram:       carousel.instagram,
    facebook:        carousel.facebook,

    angle:           carousel.angle || null,
