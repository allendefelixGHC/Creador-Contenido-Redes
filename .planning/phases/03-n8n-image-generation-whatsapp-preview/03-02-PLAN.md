```javascript
// Handle both single-post (from Normalizar URL imagen) and carousel (from Split Send WA Done)
let d;
try {
  d = $('🗂️ Collect Image URLs').first().json;
} catch(e) {
  d = $input.first().json;
}

const isCarousel = d.format === 'carousel';

const typeEmoji  = { educational:'📚', authority:'🎤', case_study:'🏆' };
const typeLabel  = { educational:'Educativo', authority:'Autoridad', case_study:'Caso de éxito' };
const modelInfo  = {
  flux:       '⚡ Flux 2 Pro',
  ideogram:   '🔤 Ideogram v3',
  nanoBanana: '🍌 Nano Banana Pro',
  custom:     '📎 Imagen propia',
};

const igPreview = d.instagram?.caption?.substring(0, 280) || 'N/A';
const fbPreview = d.facebook?.caption?.substring(0, 180) || 'N/A';
const igDots = (d.instagram?.caption?.length || 0) > 280 ? '...' : '';
const fbDots = (d.facebook?.caption?.length || 0)  > 180 ? '...' : '';

const imageStatus = isCarousel
  ? '🎠 ' + (d.image_urls?.length || 0) + ' imágenes generadas (carrusel)'
  : '🖼️ ' + (d.final_image_url ? '✅ Imagen generada' : '⚠️ Sin imagen');

const formatLine = isCarousel
  ? '\n🎠 *Formato:* Carrusel (' + d.num_images + ' slides)\n'
  : '\n';

const msg = '🚀 *PROPULSAR CONTENT ENGINE*\n\n' +
  (typeEmoji[d.type] || '📝') + ' *Tipo:* ' + (typeLabel[d.type] || d.type) + '\n' +
  '📌 *Tema:* ' + d.topic + '\n' +
  (d.angle ? '🎯 *Ángulo:* ' + d.angle + '\n' : '') +
  '📱 *Redes:* ' + d.platforms.join(', ') + '\n' +
  '🖼️ *Imagen:* ' + (modelInfo[d.image_model] || d.image_model) +
  formatLine +
  '\n━━━━━━━━━━━━━━━━\n' +
  '📸 *INSTAGRAM:*\n' + igPreview + igDots +
  '\n\n━━━━━━━━━━━━━━━━\n' +
  '👥 *FACEBOOK:*\n' + fbPreview + fbDots +
  '\n\n━━━━━━━━━━━━━━━━\n' +
  imageStatus +
  '\n\n*¿Publicar?*\n✅ *SI* para publicar\n❌ *NO* para cancelar';

return [{ json: { ...d, whatsapp_message: msg, session_id: 'propulsar_' + Date.now() } }];
```
